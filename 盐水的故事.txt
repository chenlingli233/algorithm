#include<iostream> using namespace std; int main() { 	double v, d;//有可能是小数 	int t, num; 	while (cin >> v >> d) { 		t = 0; num = 1; 		while (1) { 			for (int i = 1; i <= num; i++) { 				t++; 				v -= d; 				if (v < 0.0000001) break; 			} 			if (v < 0.0000001) break; 			t++;//停一下 			num++;停一下后连续滴的数目加1 		} 		cout << t << endl; 	} 	return 0; }
